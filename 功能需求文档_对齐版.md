## 一、移动端功能模块清单

### 1. 用户认证模块
**文件路径：** `src/pages/mobile/Login.tsx`

**功能描述：**
- 用户登录（手机号/密码、验证码）
- 忘记密码
- 用户角色区分（HQ总部/OPERATOR运营/STORE店长）

**前端数据流：**
```typescript
interface LoginParams {
  phone: string;
  password: string;
  role: 'HQ' | 'OPERATOR' | 'STORE';
}
```

**API接口需求：**
- `POST /api/v1/auth/login` - 用户登录
- `POST /api/v1/auth/logout` - 退出登录
- `GET /api/v1/auth/current` - 获取当前用户信息

**数据库表设计：**
```sql
CREATE TABLE users (
  id SERIAL PRIMARY KEY,
  phone VARCHAR(20) UNIQUE NOT NULL,
  password_hash VARCHAR(255) NOT NULL,
  role VARCHAR(20) NOT NULL CHECK (role IN ('HQ', 'OPERATOR', 'STORE')),
  store_id INTEGER REFERENCES stores(id),
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
```

### 2. 数据概览模块（Dashboard）
**文件路径：** `src/pages/mobile/Dashboard.tsx`

**功能描述：**
- 核心数据展示（评论总数、平均星级、好评率、AI回复率）
- 时间筛选（今天/昨天/7天/30天/自定义）
- 平台分布展示
- 数据源健康度监控
- 异常告警（差评紧急处理提醒）
- 门店口碑排行（HQ/OPERATOR角色）
- 门店健康值展示（STORE角色）
- 最新评论动态列表

**前端数据流：**
```typescript
interface CoreStats {
  totalReviews: number;
  reviewTrend: string;
  avgRating: number;
  ratingTrend: string;
  positiveRate: string;
  positiveTrend: string;
  aiReplyRate: string;
  replyTrend: string;
}

interface PlatformData {
  platform: string;
  count: number;
  percentage: number;
  color: string;
  icon: string;
}

interface HealthStatus {
  platform: string;
  status: 'normal' | 'delay';
  time: string;
}

interface Review {
  id: number;
  user: string;
  avatar: string;
  rating: number;
  content: string;
  platform: string;
  time: string;
  sentiment: 'positive' | 'negative' | 'neutral';
  tags: string[];
  hasImage: boolean;
}
```

**API接口需求：**
- `GET /api/v1/dashboard/core-stats?period={period}` - 获取核心统计数据
- `GET /api/v1/dashboard/platform-data` - 获取平台分布数据
- `GET /api/v1/dashboard/recent-reviews?limit={limit}` - 获取最新评论列表
- `GET /api/v1/dashboard/store-rankings?limit={limit}` - 获取门店排行榜
- `GET /api/v1/dashboard/health-status` - 获取数据源健康状态
- `GET /api/v1/dashboard/alert` - 获取异常告警

**数据库表设计：**
```sql
CREATE TABLE reviews (
  id SERIAL PRIMARY KEY,
  user_name VARCHAR(100),
  user_avatar TEXT,
  rating INTEGER CHECK (rating BETWEEN 1 AND 5),
  content TEXT,
  platform VARCHAR(50),
  platform_id VARCHAR(100),
  store_id INTEGER REFERENCES stores(id),
  sentiment VARCHAR(20),
  tags TEXT[],
  has_image BOOLEAN DEFAULT FALSE,
  images TEXT[],
  reply TEXT,
  reply_time TIMESTAMP,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_reviews_store_id ON reviews(store_id);
CREATE INDEX idx_reviews_platform ON reviews(platform);
CREATE INDEX idx_reviews_created_at ON reviews(created_at);
```

**功能描述：**

```typescript
interface NegativeReplyTask {
```

```sql
CREATE TABLE negative_reply_tasks (

  risk_level VARCHAR(20),
  scores JSONB,
  status VARCHAR(20) DEFAULT 'pending',

  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
```

## 五、数据库设计总览

### 核心表结构

```sql
-- 用户表
CREATE TABLE users (
  id SERIAL PRIMARY KEY,
  phone VARCHAR(20) UNIQUE NOT NULL,
  password_hash VARCHAR(255) NOT NULL,
  role VARCHAR(20) NOT NULL CHECK (role IN ('HQ', 'OPERATOR', 'STORE')),
  store_id INTEGER REFERENCES stores(id),
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 门店表
CREATE TABLE stores (
  id SERIAL PRIMARY KEY,
  name VARCHAR(100) NOT NULL,
  address TEXT,
  phone VARCHAR(20),
  business_hours VARCHAR(50),
  score DECIMAL(2,1),
  health INTEGER,
  platforms TEXT[],
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 评论表
CREATE TABLE reviews (
  id SERIAL PRIMARY KEY,
  user_name VARCHAR(100),
  user_avatar TEXT,
  rating INTEGER CHECK (rating BETWEEN 1 AND 5),
  content TEXT,
  platform VARCHAR(50),
  platform_id VARCHAR(100),
  store_id INTEGER REFERENCES stores(id),
  sentiment VARCHAR(20),
  tags TEXT[],
  has_image BOOLEAN DEFAULT FALSE,
  images TEXT[],
  reply TEXT,
  reply_time TIMESTAMP,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_reviews_store_id ON reviews(store_id);
CREATE INDEX idx_reviews_platform ON reviews(platform);
CREATE INDEX idx_reviews_created_at ON reviews(created_at);

-- 差评处理任务表
CREATE TABLE negative_reply_tasks (
  id SERIAL PRIMARY KEY,
  review_id INTEGER REFERENCES reviews(id),
  ai_draft TEXT,
  risk_level VARCHAR(20),
  scores JSONB,
  status VARCHAR(20) DEFAULT 'pending',
  processed_at TIMESTAMP,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 高质好评表
CREATE TABLE high_quality_reviews (
  id SERIAL PRIMARY KEY,
  review_id INTEGER REFERENCES reviews(id),
  suggested_script TEXT,
  authorized BOOLEAN DEFAULT FALSE,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 菜品表
CREATE TABLE dishes (
  id SERIAL PRIMARY KEY,
  name VARCHAR(100) NOT NULL,
  store_id INTEGER REFERENCES stores(id),
  positive_count INTEGER DEFAULT 0,
  negative_count INTEGER DEFAULT 0,
  score DECIMAL(2,1),
  type VARCHAR(20),
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 服务案例表
CREATE TABLE service_cases (
  id SERIAL PRIMARY KEY,
  type VARCHAR(20) CHECK (type IN ('golden', 'negative')),
  content TEXT,
  store_id INTEGER REFERENCES stores(id),
  status VARCHAR(20),
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 设置表
CREATE TABLE settings (
  id SERIAL PRIMARY KEY,
  key VARCHAR(100) UNIQUE NOT NULL,
  value JSONB,
  store_id INTEGER REFERENCES stores(id),
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
```
